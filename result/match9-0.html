#define NUM_PCB 30

class GradeEnvironment : public testing::Environment {
    public:
        virtual void SetUp() {
            score = 0;
            total = 160;
        }
        virtual void TearDown() {
            ::testing::Test::RecordProperty("points_given", score);
            ::testing::Test::RecordProperty("points_total", total);

        }
};
